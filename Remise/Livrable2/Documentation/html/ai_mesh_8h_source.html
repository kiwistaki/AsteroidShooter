156  const aiFace& operator = ( const aiFace& o)
157  {
158  if (&o == this)
159  return *this;
160 
161  delete[] mIndices;
162  mNumIndices = o.mNumIndices;
163  mIndices = new unsigned int[mNumIndices];
164  ::memcpy( mIndices, o.mIndices, mNumIndices * sizeof( unsigned int));
165  return *this;
166  }
167 
170  bool operator== (const aiFace& o) const
171  {
172  if (mIndices == o.mIndices)return true;
173  else if (mIndices && mNumIndices == o.mNumIndices)
174  {
175  for (unsigned int i = 0;i < this->mNumIndices;++i)
176  if (mIndices[i] != o.mIndices[i])return false;
177  return true;
178  }
179  return false;
180  }
181 
184  bool operator != (const aiFace& o) const
185  {
186  return !(*this == o);
187  }
